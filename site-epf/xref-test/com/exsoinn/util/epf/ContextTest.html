479     @Test
480     public void searchWhenTargetFieldsAreRegularExpressions() {
481         /*
482          * Data is an array of rows (I.e. array of other complex structures)
483          * Note: Escaping the "\s" like that is necessary, otherwise GSON API builds empty ("{}") object.

486                 + "{\"key\": \"Micro.*\"},"
487                 + "{\"key\": \".*gle\"},"
488                 + "{\"key\": \"Oracle.*tion\"},"
489                 + "{\"key\": [\".acebook\", \"Testsoft\" , \"Time..arner\", \"Verizon\\\\s{2}Wireless\", \"Symantec\"]},"
490                 + "{\"key\": \"8|9|10\"}"
491                 + "]}";
492         // Test anchoring partial match at end
493         Map<String, String> params = new HashMap<>();
494         params.put(AbstractContext.FOUND_ELEM_VAL_IS_REGEX, "1");
495         String topElemName = "node";
496         SearchPath sp = SearchPath.valueOf(topElemName);
497         Filter f = Filter.valueOf("key=Micro");
498         Context c = ContextFactory.INSTANCE.obtainContext(data);
499         SearchResult sr = c.findElement(sp, f, null, params);
500         assertTrue(sr.size() == 1);
501         assertTrue(sr.get(topElemName).asArray().get(0).memberValue("key").stringRepresentation().equals("Micro.*"));
